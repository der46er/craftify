// --- DYNAMISCHES NEWS SYSTEM MIT NEUEM DATUMSFORMAT --- async function loadNews() { const repo = 'der46er/craftify'; const folder = 'assets/news'; const container = document.getElementById('news-container'); // Hilfsfunktion: Wandelt "28-01-2024 19:36.txt" in ein echtes Datum um für die Sortierung function parseNewsDate(filename) { const rawName = filename.replace('.txt', ''); const parts = rawName.split(' '); const datePart = parts[0] || ''; const timePart = parts[1] ? parts[1].replace('-', ':') : '00:00'; const dParts = datePart.split('-'); if(dParts.length === 3) { // Wandelt DD-MM-YYYY in YYYY-MM-DD um, damit JavaScript es versteht return new Date(`${dParts[2]}-${dParts[1]}-${dParts[0]}T${timePart}:00`); } return new Date(0); // Fallback } try { const response = await fetch(`https://api.github.com/repos/${repo}/contents/${folder}`); if (!response.ok) throw new Error('News-Ordner nicht gefunden'); let files = await response.json(); // Filtern & Sortieren (Neuestes Datum zuerst mit der neuen Hilfsfunktion) files = files.filter(f => f.name.endsWith('.txt')).sort((a, b) => { return parseNewsDate(b.name) - parseNewsDate(a.name); }); if (files.length === 0) { container.innerHTML = `
No news articles found.
`; return; } container.innerHTML = ''; for (const file of files) { const textRes = await fetch(file.download_url); const text = await textRes.text(); const lines = text.split('\n'); const title = lines[0] || 'Update'; let rawContent = lines.slice(1).join('\n').trim(); // Bild-Formatierung const imgRegex = /!\[(.*?)\]\((.*?)\)/g; rawContent = rawContent.replace(imgRegex, ''); const content = rawContent.replace(/\n/g, '
'); // Anzeige für die Webseite formatieren const dateObj = parseNewsDate(file.name); const rawName = file.name.replace('.txt', ''); const rawTime = rawName.split(' ')[1] || ''; let displayDate = rawName; let displayTime = ''; if (dateObj.getTime() !== 0) { const isDe = document.documentElement.lang === 'de'; displayDate = dateObj.toLocaleDateString(isDe ? 'de-DE' : 'en-GB', { day: '2-digit', month: '2-digit', year: 'numeric' }); if (rawTime) { // Ersetzt eventuelle Bindestriche wieder durch Doppelpunkte const formattedTime = rawTime.replace('-', ':'); displayTime = ` • ${formattedTime} ${isDe ? 'Uhr' : ''}`; } } // Karte bauen const card = document.createElement('div'); card.className = 'game-card p-6 md:p-10 rounded-[2rem] flex flex-col gap-4'; card.innerHTML = `
${displayDate}${displayTime}

${title}

${content}

`; container.appendChild(card); } lucide.createIcons(); } catch (error) { console.error(error); container.innerHTML = `
Failed to load news.
`; } }
